The Engine Code (src/manifest_parser.rs)This structure mirrors your "Sovereign" logic, allowing you to toggle the Tironian Cipher or RTX 3090 optimizations via the GUI.Rustuse serde::Deserialize;
use std::collections::HashMap;

#[derive(Debug, Deserialize)]
pub struct FactoryManifest {
    pub factory: FactoryConfig,
    pub profiles: HashMap<String, Profile>,
    pub nuke_list: NukeList,
}

#[derive(Debug, Deserialize)]
pub struct FactoryConfig {
    pub target_build: String,
    pub wimlib_path: String,
    pub reg_editor_path: String,
}

#[derive(Debug, Deserialize)]
pub struct Profile {
    pub friendly_name: String,
    pub base_index: i32,
    pub features: Features,
    pub drivers: Option<HashMap<String, String>>,
    pub secrets: Option<SecretConfig>,
}

#[derive(Debug, Deserialize)]
pub struct Features {
    pub xbox_mode: bool,
    pub ai_suppression: String, // "Soft", "Hard", "Total"
    pub tpm_bypass: bool,
}

#[derive(Debug, Deserialize)]
pub struct SecretConfig {
    pub layout: String, // e.g., "Tironian-Notes"
    pub font: String,
}

#[derive(Debug, Deserialize)]
pub struct NukeList {
    pub packages: Vec<String>,
}
